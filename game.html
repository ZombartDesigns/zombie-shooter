const config = {

type: Phaser.AUTO,

width: 800,

height: 600,

physics: {

default: 'arcade',

arcade: {

gravity: { y: 0 },

debug: false,

},

},

scene: {

preload: preload,

create: create,

update: update,

},

};


const game = new Phaser.Game(config);


let player;

let zombies;

let bullets;

let cursors;

let spaceKey;

let currentLevel = 1;

let zombieSpeed = 200; // Initial speed

let zombieSpawnRate = 1000; // Initial spawn rate in milliseconds

let levelDuration = 120; // Level duration in seconds

let levelStartTime;

let score = 0;

let scoreText;


function preload() {

// Load placeholder images

this.load.image('player', 'https://via.placeholder.com/50/FFFFFF/000000?text=P');

this.load.image('zombie', 'https://via.placeholder.com/50/FF0000/FFFFFF?text=Z');

this.load.image('bullet', 'https://via.placeholder.com/5/FFFFFF/FFFFFF');

}


function create() {

player = this.physics.add.image(400, 550, 'player').setCollideWorldBounds(true);

zombies = this.physics.add.group();

bullets = this.physics.add.group();


cursors = this.input.keyboard.createCursorKeys();

spaceKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE);


scoreText = this.add.text(16, 16, 'Score: 0', { fontSize: '32px', fill: '#fff' });


levelStartTime = this.time.now;

this.time.addEvent({ delay: zombieSpawnRate, callback: spawnZombie, callbackScope: this, loop: true });

}


function update() {

if (cursors.left.isDown) {

player.setVelocityX(-300);

} else if (cursors.right.isDown) {

player.setVelocityX(300);

} else {

player.setVelocityX(0);

}


if (spaceKey.isDown) {

shootBullet();

}


checkLevelUp();

}


function shootBullet() {

const bullet = bullets.create(player.x, player.y - 20, 'bullet');

bullet.setVelocityY(-500);

}


function spawnZombie() {

const zombieX = Phaser.Math.Between(0, config.width);

const zombie = zombies.create(zombieX, 0, 'zombie');

zombie.setVelocityY(zombieSpeed);

}


function checkLevelUp() {

if (this.time.now - levelStartTime > levelDuration * 1000) {

currentLevel++;

zombieSpeed += 50; // Increase zombie speed

zombieSpawnRate = Math.max(500, zombieSpawnRate - 50); // Increase spawn rate

this.time.addEvent({ delay: zombieSpawnRate, callback: spawnZombie, callbackScope: this, loop: true });

levelStartTime = this.time.now;

updateScore();

}


// Check for collisions

this.physics.overlap(bullets, zombies, hitZombie, null, this);

}


function hitZombie(bullet, zombie) {

bullet.destroy();

zombie.destroy();

score += 10;

updateScore();

}


function updateScore() {

scoreText.setText(`Score: ${score}`);

}
